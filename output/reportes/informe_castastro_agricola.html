frutales_rendimiento %>%
  group_by(nom_region, ao, especie) %>%
  summarise(rendimiento_ton_ha=sum(rendimiento_ton_ha, na.rm = T)) %>%
  filter(especie %in% c("kiwi", "cerezo")) %>%
  filter(ao>year("1999-01-01")) %>%
  ggplot(aes(ao,rendimiento_ton_ha, fill=especie)) +
    geom_col(position = "dodge")+
    facet_wrap(~nom_region)+
    theme_bw()+
    theme(axis.text.x =   element_text(angle=90),
          legend.position = "bottom")
## `summarise()` has grouped output by 'nom_region', 'ao'. You can override using the `.groups` argument.
